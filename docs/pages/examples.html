Examples

Practical patterns for common integrations.

Batch

import { processTexts } from 'qirrel';

const results = await processTexts([
  'Email: team@example.com',
  'Call: +44 20 7946 0958'
], undefined, { concurrency: 2 });

Custom Processor

const hashtagProcessor = {
  name: 'hashtag-extractor',
  cacheable: true,
  async run(ctx) {
    return ctx;
  }
};
